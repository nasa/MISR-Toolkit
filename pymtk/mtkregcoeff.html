3.8 MtkRegCoeff

class MtkRegCoeff

>>> regr_coeff = MtkRegCoeff()


>>> regr_coeff, regr_mapinfo = regr.coeff_calc(data1, mask1, data2, sigma2, mask2, sizefactor, mapinfo)
